\begin{align}
\myvec{x\\-y}=\myvec{-6&-24\\24&-6}{\myvec{3&-5\\5&3}}^{-1}\myvec{1\\0}\\
=\frac{1}{34}\myvec{-6&-24\\24&-6}\myvec{3&5\\-5&3}\myvec{1\\0}\\
=\frac{1}{34}\myvec{102&-102\\102&102}\myvec{1\\0}\\

\implies\myvec{x\\-y}=\myvec{3\\3}\\
