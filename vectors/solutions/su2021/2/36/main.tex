Let
\begin{align}
    \vec{n_1} &= \myvec{4 \\ 8} - \myvec{-2 \\ 6}\\
    &= \myvec{6 \\ 2}
\end{align}
and
\begin{align}
     \vec{n_2} &= \myvec{x \\ 24} - \myvec{8 \\ 12}\\
    &= \myvec{x - 8 \\ 12}
\end{align}
From the given information, 
\begin{align}
 \vec{n_1^\top}\vec{n_2}  &= 0\\
 \implies  \myvec{6 & 2} \myvec{x - 8 \\ 12}&=0
 \\
 \text{or, }    x = 4
\end{align}
