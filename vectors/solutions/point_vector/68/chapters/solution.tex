
\begin{equation}
\begin{aligned}
\norm{\vec{a} - \vec{b}}^2 = \norm{\vec{a}}^2 + \norm{\vec{b}}^2 - 2\vec{a}^T \vec{b}\\
\implies \norm{\vec{a} - \vec{b}}^2 = 2^2 + 3^2 - 2\times 4\\
\implies \norm{\vec{a} - \vec{b}}^2 = 5\\
\implies \norm{\vec{a} - \vec{b}} = \sqrt{5}\\
\end{aligned}
\end{equation}
