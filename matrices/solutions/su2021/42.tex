\begin{enumerate}
\item Let

\begin{align}
\vec{A}^{\top}=\vec{A}
\label{matrix/42/sym}
\end{align}
then
\begin{align}
\brak{\vec{B}^{\top}\vec{A}\vec{B}}^\top
&=\vec{B}^{\top}\vec{A}^{\top}\vec{B}\\
&=\vec{B}^{\top}\vec{A}\vec{B}\\
\end{align}
using \ref{matrix/42/sym}
Hence
\begin{align}
\vec{B}^{\top}\vec{A}\vec{B}
\end{align}
is symmetric. 
\item If 
\begin{align}
\vec{A}^\top=-\vec{A}
\label{matrix/42/asym}
\end{align}
then
\begin{align}
    \brak{\vec{B}^{\top}\vec{A}\vec{B}}^\top
    &=\vec{B}^{\top}\vec{A}^{\top}\vec{B}\\
    &=-\vec{B}^{\top}\vec{A}\vec{B}\\
    \end{align}
    \begin{align}
\therefore \vec{B}^{\top}\vec{A}\vec{B}
\end{align}
is skew symmetric. 
\end{enumerate}
