Let
\begin{align}
    A=\myvec{x & z \\ y & t} \label{aug/2/10eq:A}
\end{align}
Now,
\begin{align}
    &2 \vec{A} +  \myvec{3 & -3 \\ 0 & 6} =  \myvec{9 & 15 \\ 12 & 18} \\
   \implies &2 \vec{A}  =  \myvec{9-3 & 15+3 \\ 12-0 & 18-6}\\
   \implies & \vec{A}  = \frac{1}{2} \myvec{6 & 18 \\ 12 & 12}\\
   \implies &\vec{A} = \myvec{3 & 9 \\ 6 & 6}   
\end{align}
