    \item The direction vectors of the lines are :
\begin{align}
    \vec{m_1}=\myvec{3 \\ 2 \\6} \\    
    \vec{m_2}=\myvec{1 \\ 2 \\2}
\end{align}
Thus, the angle $\theta$ between two vectors is given by 
\begin{align}
\cos \theta &= \frac{\vec{m_1}^T\vec{m_2}}{\norm{\vec{m_1}}\norm{\vec{m_2}}}\\
&=\frac{19}{3\times7}\\
\implies \theta &= 25.21\degree
\end{align}
\item The direction vectors of the lines are:
\begin{align}
    \vec{m_1}=\myvec{1 \\ -1 \\-2}\\
    \vec{m_2}=\myvec{3 \\ -5 \\-4}.
\end{align}
Thus, the angle $\theta$ between two vectors is given by 
\begin{align}
\cos \theta &= \frac{\vec{m_1}^T\vec{m_2}}{\norm{\vec{m_1}}\norm{\vec{m_2}}}\\
