\implies \myvec{-a & 3}\vec{x} &= 7

\begin{align}
\myvec{-a & 3}\myvec{3\\4} &= 7
\\
\implies -3a +12 &= 7
\\
\implies a &= \frac{5}{3}
\end{align}
