\begin{equation}
 \vec n_1=\myvec{2\\2\\-3},
 \vec n_2 =\myvec{3\\-3\\5},
\end{equation}

Let $\theta$ be angle between vectors $\vec{n}_1$,$\vec{n}_2$.  Then,
\begin{align}
    \theta &= \cos^{-1}\brak{{\frac{\vec{n}_1^{T}\vec{n}_2}{\norm{\vec{n_1}}\norm{\vec{n}_2}}}}
 &=\cos^{-1}\brak{-\frac{1}{2}}\\
\implies \theta &= 120\degree
\end{align}
