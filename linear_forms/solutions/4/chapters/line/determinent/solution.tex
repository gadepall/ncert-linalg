\begin{align}
\mydet{3\vec{A}} = \mydet{ 
3 & 0 & 3 
\\
0 & 3 & 6
\\
0 & 0 & 12
}
= 108
\end{align}
and 
\begin{align}
\mydet{\vec{A}} = 4
\end{align}
Hence, 
\begin{align}
\mydet{3\vec{A}} = 27\mydet{\vec{A}}
\end{align}
In general, for an $n\times n$ matrix $\vec{A}$, 
\begin{align}
\mydet{k\vec{A}} = k^n\mydet{\vec{A}}
\end{align}
